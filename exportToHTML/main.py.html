2    import sys, pygame

4    pygame.init() 
5    size = width, height = 600, 600 
6    black = 0, 0, 0 
7    white = 255, 255, 255 
8    screen = pygame.display.set_mode(size) 
9    pygame.display.set_caption("Assignment: Ball Game") 
10   ball = pygame.image.load("fireball.png") 
11   pygame.key.set_repeat(10, 10) 
12   clk = pygame.time.Clock() 
13   loop = True 
14   x, y = 0, 0 
15   imgx, imgy = 0, 0 
16    
17    
18   def message_display(text): 
19       txt = pygame.font.Font('freesansbold.ttf', 20) 
20       s = txt.render(text, True, black) 
21       sr = s.get_rect() 
22       sr.center = (400, 10) 
23       screen.blit(s, sr) 
24    
25    
26   while loop: 
27       for event in pygame.event.get(): 
28           if event.type == pygame.QUIT: 
29               loop = False 
30           if event.type == pygame.KEYDOWN: 
31               if event.key == pygame.K_LEFT: 
32                   x = -5 
33               elif event.key == pygame.K_UP: 
34                   y = -5 
35               elif event.key == pygame.K_RIGHT: 
36                   x = 5 
37               elif event.key == pygame.K_DOWN: 
38                   y = 5 
39           if event.type == pygame.KEYUP: 
40               if event.key == pygame.K_LEFT or event.key == pygame.K_RIGHT or event.key == pygame.K_UP or event.key == pygame.K_DOWN: 
41                   x = 0 
42                   y = 0 
43       imgx += x 
44       imgy += y 
45       screen.fill(white) 
46       screen.blit(ball, (imgx, imgy)) 
47       text = "x: " + str(imgx) + "  y: " + str(imgy) 
48       message_display(text) 
49       pygame.display.update() 
50       clk.tick(60)
